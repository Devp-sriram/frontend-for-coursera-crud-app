**RBAC Middleware Example:**
```ts
export function authorizeRole(roles: string[]) {
  return function (req, res, next) {
    const userRole = req.user.role;
    if (!roles.includes(userRole)) return res.status(403).json({ message: "Forbidden" });
    next();
  };
}
```
